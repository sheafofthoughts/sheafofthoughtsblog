\begin{prop}[]
	\label{prop:vanishingofcohomologyirreducible}
	Let $ X $ be an affine irreducible variety, $ \{Q_i\}_{i\in I} $ a finite family of regular functions on $ X $ that do not vanish simultaneously, and $ \mathfrak{U} $ the open covering of $ X $ consisting of $ X _{Q_i}=U_i $. If $ \mathcal{F} $ is a coherent algebraic subsheaf of $ \mathcal{O}_X^p $, then $ \mathrm{H}^q(\mathfrak{U}, \mathcal{F})=0 $ for all $ q>0 $.
\end{prop}
\begin{proof}
We can assume that $ U_i\neq \emptyset $ for all $ i $. Let $ f=(f _{i_0,\dots , i_q}) $ be a $ q $-cocycle of $ \mathfrak{U} $ with values in $ \mathcal{F} $. Every $ f _{i_0,\dots, i_q}\in \mathcal{F}(U _{i_0\dots ,i_q}) $ can be identified with a system of $ p $ regular functions on $ U _{i_0,\dots , i_q} $; we can apply proposition \ref{prop:regularfunctiononX} to $ Q:=Q _{i_0}\cdots Q _{i_q} $ and get, for $ n $ sufficiently large, a system of $ p $ regular functions on $ X $ identifying $ g _{i_0,\dots , i_q}=Q^n f _{i_0,\dots ,i_q} $. Since the covering is finite, we can consider an $ n $ such that $ Q^n f _{i_0,\dots ,i_q} $ is identified with $ p $ regular functions on $ X $ for every $ i_0,\dots , i_q $. Hence, the image of $ g _{i_0,\dots , i_q} $ in $ (\mathcal{O}_X^p/\mathcal{F})( U _{i_0,\dots , i_q}) $ is zero; therefore, we can apply proposition \ref{prop:zeroonthewholeX} and get that $Q^m g _{i_0,\dots, i_q} $ is zero on the whole $ (\mathcal{O}_X^p/\mathcal{F})(X) $ for some $ m\in \mathbb{N} $. Define $ N:=n+m $, we get a section $ h _{i_0,\dots , i_q} \in \mathcal{F}(X) $ coinciding with $ Q^N f _{i_0,\dots , i_q} $ on $ U _{i_0,\dots ,i_q} $. 

Since the $ Q_i^N $ do not vanish simultaneously, there exist regular functions $ R_i\in \mathcal{O}_X(X) $ such that $ \sum R_i Q_i^N=1 $. We can define for any system $ i_0,\dots ,i_q $ the section
\[
	k _{i_0,\dots , i _{q-1}}:=\sum_i R_i\cdot h _{i,i_0,\dots, i _{q-1}}/(Q _{i_0}\cdots Q _{i _{q-1}})^N
\]
which defines a cochain $ k\in C ^{q-1}(\mathfrak{U},\mathcal{F}) $. The claim is now that $ f=dk $, which is a simple check. The key observation is that it suffices to show that $ f _{i_0,\dots i _{q-1}}=(dk) _{i_0,\dots , i _{q-1}} $ on $ \bigcap_i U_i $; the equality on $ U _{i_0,\dots ,i_q} $ follows because we can identify these sections with systems of $ p $ rational functions on $ X $. % thanks to proposition \ref{prop:isomorphismquotientringglobalsection}
\end{proof}

\begin{corollary}[]
	$ \mathrm{H}^q(X,\mathcal{F})=0 $ for $ q>0 $.
\end{corollary}
\begin{corollary}[]
	\label{corl:homoissurj}
	The homomorphism $ \mathcal{O}_X^p(X)\to (\mathcal{O}_X^p/\mathcal{F})(X) $ is surjective.
\end{corollary}

We can also generalize proposition \ref{prop:isomorphismquotientringglobalsection} by removing the hypothesis of irreducibility.
\begin{corollary}[]
	\label{corl:functionsarepolynomials}
	Let $ V $ be a closed subvariety of $ K^r $ and $ A := K[X_1,\dots ,X_r]/\mathrm{I}(V) $. The homomorphism $ \iota: A\to \mathcal{O}_V(V) $ is bijective.
\end{corollary}
\begin{proof}
	Apply corollary \ref{corl:homoissurj} with $ X:=K^r $, $ p=1 $, $ \mathcal{F}=\mathcal{I}_V $ sheaf of ideals defined by $ V $. 
\end{proof}
